126  if (m_dataFormat == NONCONTIGUOUS || m_isConst) {

131  return false;
132  }
133 
134  if (typeid(getBuffer<T>()).hash_code() != typeid(T*).hash_code()) {
135  return false;
136  }
137 
138  switch (m_dataFormat) {
139  case CONTIGUOUS:
140  // Delete the current array
141  if (m_bufferPtr) {

144  m_width = 0;
145  }
146  // Allocate a new empty array

148  m_width = length;
